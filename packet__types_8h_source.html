185static inline void packet_header_set_length(PacketHeader *p, const uint16_t length) {
186 uint8_t encoded_length = ((length + sizeof(PacketHeader)) / 4) - 1; // Include header length, 4 byte increments
187 p->bytes[6] &= ~0xFC; // Clear top 6 bits
188 p->bytes[6] |= (encoded_length & 0x3F) << 2; // OR in bottom 6 bits of length, shifted left
189}

196static inline uint16_t packet_header_get_length(const PacketHeader *p) { return (((p->bytes[6] & 0xFC) >> 2) + 1) * 4; }

203static inline void block_header_set_length(BlockHeader *b, const uint16_t length) {
204 uint16_t encoded_length = ((length + sizeof(BlockHeader)) / 4) - 1; // Add header size, 4 byte increments
205 *(uint32_t *)(&b->bytes[0]) &= ~0x1F; // Clear the lowest 5 bits
206 *(uint32_t *)(&b->bytes[0]) |= encoded_length & 0x1F; // Last 5 bits are length
207}
